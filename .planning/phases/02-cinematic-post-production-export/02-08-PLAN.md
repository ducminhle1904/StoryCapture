    ```rust
    pub enum ClickSfxLevel { Off, Subtle, Pronounced }
    pub fn click_gain(level: ClickSfxLevel) -> f32 { match level { ClickSfxLevel::Off => 0.0, ClickSfxLevel::Subtle => 0.3, ClickSfxLevel::Pronounced => 0.7 } }
    pub struct ClickEvent { pub t_ms: u64, pub sfx_file: String /* id in manifest */ }
    pub fn emit_click_sfx(events: &[ClickEvent], level: ClickSfxLevel, sound_root: &Path, input_index_start: usize) -> (String, Vec<ExtraInput>) {
        if matches!(level, ClickSfxLevel::Off) || events.is_empty() { return (String::new(), vec![]); }
        let gain = click_gain(level);
        let mut extra_inputs = Vec::new();
        let mut segments = Vec::new();
        let mut labels = Vec::new();
        for (i, ev) in events.iter().enumerate() {
            let idx = input_index_start + i;
            extra_inputs.push(ExtraInput::audio_file(sound_root.join("sfx").join(&ev.sfx_file)));
            let seg_label = format!("[sfx_{i}]");
            segments.push(format!("[{idx}:a]volume={gain},adelay={t}|{t}{seg}", idx = idx, gain = gain, t = ev.t_ms, seg = seg_label));
            labels.push(seg_label);
        }
        let merge_inputs: String = labels.join("");
        let merge = format!("{merge_inputs}amix=inputs={n}:duration=longest:normalize=0[sfx_mix]", merge_inputs = merge_inputs, n = labels.len());
        (segments.join(";") + ";" + &merge, extra_inputs)
    }
    ```
