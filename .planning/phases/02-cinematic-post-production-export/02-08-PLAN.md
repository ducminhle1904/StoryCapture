    ```rust
    pub enum ClickSfxLevel { Off, Subtle, Pronounced }
    pub fn click_gain(level: ClickSfxLevel) -> f32 { match level { ClickSfxLevel::Off => 0.0, ClickSfxLevel::Subtle => 0.3, ClickSfxLevel::Pronounced => 0.7 } }
    pub struct ClickEvent { pub t_ms: u64, pub sfx_file: String /* id in manifest */ }
    pub fn emit_click_sfx(events: &[ClickEvent], level: ClickSfxLevel, sound_root: &Path, input_index_start: usize) -> (String, Vec<ExtraInput>) {
        if matches!(level, ClickSfxLevel::Off) || events.is_empty() { return (String::new(), vec![]); }
        let gain = click_gain(level);
        let mut extra_inputs = Vec::new();
        let mut segments = Vec::new();
        let mut labels = Vec::new();
        for (i, ev) in events.iter().enumerate() {
            let idx = input_index_start + i;
            extra_inputs.push(ExtraInput::audio_file(sound_root.join("sfx").join(&ev.sfx_file)));
            let seg_label = format!("[sfx_{i}]");
            segments.push(format!("[{idx}:a]volume={gain},adelay={t}|{t}{seg}", idx = idx, gain = gain, t = ev.t_ms, seg = seg_label));
            labels.push(seg_label);
        }
        let merge_inputs: String = labels.join("");
        let merge = format!("{merge_inputs}amix=inputs={n}:duration=longest:normalize=0[sfx_mix]", merge_inputs = merge_inputs, n = labels.len());
        (segments.join(";") + ";" + &merge, extra_inputs)
    }
    ```

    ```rust
    // Produces silent WAV/OGG placeholders for every manifest entry.
    // Replace with real sourced files during curation (scripts/curate-sound-library.md).
    use hound::{WavSpec, WavWriter, SampleFormat};
    fn main() {
        let manifest: serde_json::Value = serde_json::from_str(include_str!("../../../assets/sound-library/manifest.json")).unwrap();
        for entry in manifest["entries"].as_array().unwrap() {
            let cat = entry["category"].as_str().unwrap();
            let file = entry["file"].as_str().unwrap();
            let dur_ms = entry["duration_ms"].as_u64().unwrap();
            let path = Path::new("assets/sound-library").join(cat).join(file);
            std::fs::create_dir_all(path.parent().unwrap()).unwrap();
            if file.ends_with(".wav") {
                let spec = WavSpec { channels: 1, sample_rate: 48000, bits_per_sample: 16, sample_format: SampleFormat::Int };
                let mut w = WavWriter::create(&path, spec).unwrap();
                for _ in 0..((48000 * dur_ms / 1000) as usize) { w.write_sample(0i16).unwrap(); }
                w.finalize().unwrap();
            } else if file.ends_with(".ogg") {
                // Use `ogg` + `vorbis-encoder` crates; if too heavy, fall back to shelling out to the Phase 1 ffmpeg sidecar:
                // ffmpeg -f lavfi -i anullsrc=r=48000:cl=stereo -t {dur_sec} -c:a libvorbis -b:a 96k {path}
                run_ffmpeg_silence_ogg(&path, dur_ms);
            }
        }
    }
    ```
