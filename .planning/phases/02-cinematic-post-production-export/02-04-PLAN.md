    ```typescript
    import type { PreviewBackend } from './types';
    export type PreviewBackend = 'webgpu' | 'webgl2';
    export type PreviewCtx =
      | { kind: 'webgpu'; device: GPUDevice; context: GPUCanvasContext; format: GPUTextureFormat; adapter: GPUAdapter }
      | { kind: 'webgl2'; gl: WebGL2RenderingContext };

    export async function initPreviewContext(canvas: HTMLCanvasElement): Promise<PreviewCtx> {
      if (typeof navigator !== 'undefined' && 'gpu' in navigator && navigator.gpu) {
        const adapter = await navigator.gpu.requestAdapter();
        if (adapter) {
          const device = await adapter.requestDevice();
          const context = canvas.getContext('webgpu') as GPUCanvasContext | null;
          if (context) {
            const format = navigator.gpu.getPreferredCanvasFormat();
            context.configure({ device, format, alphaMode: 'premultiplied' });
            console.info('[preview] using WebGPU');
            return { kind: 'webgpu', device, context, format, adapter };
          }
        }
      }
      const gl = canvas.getContext('webgl2') as WebGL2RenderingContext | null;
      if (gl) {
        console.info('[preview] using WebGL2 fallback');
        return { kind: 'webgl2', gl };
      }
      throw new Error('Neither WebGPU nor WebGL2 available — preview unsupported');
    }
    ```

    ```typescript
    export class PreviewEngine {
      private ctx: PreviewCtx | null = null;
      private backendImpl: WebGPUBackend | WebGL2Backend | null = null;
      constructor(private config: PreviewEngineConfig) {}
      async init(): Promise<void> {
        this.ctx = await initPreviewContext(this.config.canvas);
        if (this.ctx.kind === 'webgpu') this.backendImpl = new WebGPUBackend(this.ctx, this.config);
        else this.backendImpl = new WebGL2Backend(this.ctx, this.config);
        await this.backendImpl.init();
      }
      async renderFrame(t_ms: number, plan: PreviewRenderPlan): Promise<void> {
        if (!this.backendImpl) throw new Error('PreviewEngine not initialised');
        // Acquire current video frame from <video> element via requestVideoFrameCallback or copy via GPUExternalTexture.
        // For the stub: seek video to t_ms/1000 and render when seeked.
        return this.backendImpl.renderFrame(t_ms, plan);
      }
      dispose(): void { this.backendImpl?.dispose(); this.ctx = null; this.backendImpl = null; }
      get backend(): PreviewBackend { return this.ctx?.kind ?? 'webgl2'; }
    }
    ```
