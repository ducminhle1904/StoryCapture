    ```typescript
    export interface HistoryEntry { action: UndoableAction; appliedAt: number }
    export class HistoryBuffer {
      private entries: HistoryEntry[] = [];
      private cursor = -1;
      constructor(private cap: number = 50) {}
      push(entry: HistoryEntry): void {
        // Truncate redo branch
        if (this.cursor < this.entries.length - 1) this.entries = this.entries.slice(0, this.cursor + 1);
        this.entries.push(entry);
        if (this.entries.length > this.cap) this.entries.shift();
        this.cursor = this.entries.length - 1;
      }
      canUndo(): boolean { return this.cursor >= 0; }
      canRedo(): boolean { return this.cursor < this.entries.length - 1; }
      popUndo(): HistoryEntry | null { if (!this.canUndo()) return null; return this.entries[this.cursor--]; }
      popRedo(): HistoryEntry | null { if (!this.canRedo()) return null; return this.entries[++this.cursor]; }
      get length(): number { return this.entries.length; }
      get cursorPosition(): number { return this.cursor; }
      clear(): void { this.entries = []; this.cursor = -1; }
    }
    ```

    ```typescript
    export const COALESCE_IDLE_MS = 500;
    export type CoalesceKey = string; // e.g. "move-clip:<clipId>" or "set-effect-param:<nodePath>:<field>"
    export function coalesceKey(a: UndoableAction): CoalesceKey | null {
      switch (a.kind) {
        case 'move-clip':        return `move-clip:${a.clipId}`;
        case 'trim-clip':        return `trim-clip:${a.clipId}`;
        case 'set-effect-param': return `set-effect-param:${a.nodePath}:${a.field}`;
        case 'edit-text-overlay':return `edit-text-overlay:${a.overlayId}`;
        // discrete actions do not coalesce:
        case 'delete-clip': case 'add-sound-clip': case 'apply-preset': case 'revert-preset': case 'change-background':
          return null;
      }
    }
    export class Coalescer {
      private last: { key: CoalesceKey; entry: HistoryEntry } | null = null;
      constructor(private idleMs: number = COALESCE_IDLE_MS) {}
      /**
       * Returns `coalescedInto: HistoryEntry` when action extends the current coalesce window
       * (caller should update the TOP of the history buffer with the coalesced entry).
       * Returns `{ newEntry: HistoryEntry }` when action starts a new entry.
       */
      feed(action: UndoableAction, now: number): { kind: 'coalesced'; entry: HistoryEntry } | { kind: 'new'; entry: HistoryEntry } {
        const key = coalesceKey(action);
        if (key && this.last && this.last.key === key && (now - this.last.entry.appliedAt) <= this.idleMs) {
          const merged = mergeActions(this.last.entry.action, action);
          const entry: HistoryEntry = { action: merged, appliedAt: now };
          this.last = { key, entry };
          return { kind: 'coalesced', entry };
        }
        const entry: HistoryEntry = { action, appliedAt: now };
        this.last = key ? { key, entry } : null;
        return { kind: 'new', entry };
      }
      reset() { this.last = null; }
    }
    function mergeActions(prev: UndoableAction, next: UndoableAction): UndoableAction {
      // For move-clip: keep prev.fromMs, take next.toMs.
      // For trim-clip: keep prev.fromRange, take next.toRange.
      // For set-effect-param: keep prev.prev, take next.next.
      // For edit-text-overlay: keep prev.prev, take next.next.
      // ...
    }
    ```

    ```typescript
    import type { StateCreator } from 'zustand';
    export interface UndoSlice {
      history: HistoryBuffer;
      coalescer: Coalescer;
      canUndo: boolean;
      canRedo: boolean;
      pushAction: (action: UndoableAction) => void;
      undo: () => void;
      redo: () => void;
      clearHistory: () => void;
    }
    export const createUndoSlice: StateCreator<EditorStore, [], [], UndoSlice> = (set, get) => ({
      history: new HistoryBuffer(50),
      coalescer: new Coalescer(500),
      canUndo: false,
      canRedo: false,
      pushAction: (action) => {
        const now = performance.now();
        const result = get().coalescer.feed(action, now);
        const history = get().history;
        if (result.kind === 'coalesced') {
          // Replace top entry
          (history as any).entries[history.cursorPosition] = result.entry;
        } else {
          history.push(result.entry);
        }
        set(state => { applyAction(state, action); state.canUndo = history.canUndo(); state.canRedo = history.canRedo(); });
      },
      undo: () => {
        const entry = get().history.popUndo();
        if (!entry) return;
        const inverted = invertAction(entry.action);
        set(state => { applyAction(state, inverted); state.canUndo = state.history.canUndo(); state.canRedo = state.history.canRedo(); });
        get().coalescer.reset();
      },
      redo: () => {
        const entry = get().history.popRedo();
        if (!entry) return;
        set(state => { applyAction(state, entry.action); state.canUndo = state.history.canUndo(); state.canRedo = state.history.canRedo(); });
        get().coalescer.reset();
      },
      clearHistory: () => {
        get().history.clear();
        get().coalescer.reset();
        set(state => { state.canUndo = false; state.canRedo = false; });
      },
    });
    ```
