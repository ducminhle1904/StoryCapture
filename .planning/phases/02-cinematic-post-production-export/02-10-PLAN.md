    ```rust
    pub struct RenderQueueActor {
        pool: SidecarPool,
        db: Arc<Mutex<rusqlite::Connection>>,
        progress_tx: mpsc::Sender<RenderProgress>,
        rx: mpsc::Receiver<QueueMsg>,
        running: HashMap<Uuid, CancellationToken>,
    }
    impl RenderQueueActor {
        pub async fn init_resume(&self) -> Result<u32, EncoderError> {
            let conn = self.db.lock().await;
            storage::render_job_repo::on_startup_mark_orphans(&conn).map_err(Into::into)
        }
        pub async fn run(mut self) {
            self.init_resume().await.ok();
            loop {
                tokio::select! {
                    Some(msg) = self.rx.recv() => self.handle_msg(msg).await,
                    _ = self.try_poll_and_spawn() => {},
                    _ = tokio::time::sleep(Duration::from_millis(500)) => {},
                }
            }
        }
        async fn try_poll_and_spawn(&mut self) -> Result<(), EncoderError> {
            let capacity = self.pool.cfg.max_concurrent.saturating_sub(self.running.len());
            if capacity == 0 { return Ok(()); }
            let ready: Vec<RenderJob> = { let conn = self.db.lock().await; storage::render_job_repo::poll_ready(&conn, capacity as u32)? };
            for job in ready {
                self.spawn_job(job).await?;
            }
            Ok(())
        }
        async fn spawn_job(&mut self, job: RenderJob) -> Result<(), EncoderError> {
            let cancel = CancellationToken::new();
            self.running.insert(job.id, cancel.clone());
            { let conn = self.db.lock().await; storage::render_job_repo::mark_running(&conn, job.id)?; }
            let db = self.db.clone();
            let progress_tx = self.progress_tx.clone();
            let pool = self.pool.clone();
            tokio::spawn(async move {
                let _permit = pool.acquire().await;
                let result = execute_render_job(&job, db.clone(), progress_tx, cancel).await;
                let conn = db.lock().await;
                match result {
                    Ok(output_path) => { storage::render_job_repo::mark_completed(&conn, job.id, &output_path).ok(); }
                    Err(EncoderError::Cancelled) => { storage::render_job_repo::cancel(&conn, job.id).ok(); }
                    Err(e) => { storage::render_job_repo::mark_failed(&conn, job.id, &e.to_string()).ok(); }
                }
            });
            Ok(())
        }
        async fn handle_msg(&mut self, msg: QueueMsg) { /* cancel/shutdown */ }
    }
    pub async fn spawn_render_queue(...) -> RenderQueueHandle { /* create actor, spawn tokio task, return handle */ }
    ```

    ```rust
            tokio::spawn(async move { encode_one(cmd, interm, spec.clone(), hw).await })
        }).collect();
        let results: Vec<_> = futures::future::try_join_all(tasks).await?;
        Ok(results.into_iter().collect::<Result<Vec<_>, _>>()?)
    }
    async fn encode_one(cmd: TauriSidecarCommand, input_ffv1: PathBuf, spec: OutputSpec, hw: EncoderProbe) -> Result<PathBuf, EncoderError> {
        let mut args = vec!["-y".into(), "-i".into(), input_ffv1.to_string_lossy().into()];
        match spec.format {
            OutputFormat::Mp4 => {
                args.push("-c:v".into()); args.push(hw.best_h264_encoder().to_string());
                args.push("-b:v".into()); args.push(bitrate_for(spec.resolution, spec.quality, "h264"));
                args.push("-c:a".into()); args.push("aac".into());
                args.push("-movflags".into()); args.push("+faststart".into());
            }
            OutputFormat::WebM => {
                args.push("-c:v".into()); args.push("libvpx-vp9".into());
                args.push("-b:v".into()); args.push(bitrate_for(spec.resolution, spec.quality, "vp9"));
                args.push("-c:a".into()); args.push("libopus".into());
            }
            OutputFormat::Gif => {
                // 2-pass: palettegen then paletteuse (Pitfall #7)
                args.push("-filter_complex".into());
                args.push(format!("[0:v]fps={fps},scale={sz}:-1:flags=lanczos,split[a][b];[a]palettegen=stats_mode=full[p];[b][p]paletteuse=dither=bayer:bayer_scale=5:diff_mode=rectangle",
                    fps = spec.fps, sz = resolution_width(spec.resolution)));
            }
        }
        if let Resolution::R720p | Resolution::R1080p | Resolution::R4k = spec.resolution { /* add -vf scale */ }
        args.push(spec.output_path.to_string_lossy().into());
        let sidecar = FfmpegSidecar::spawn(cmd, args).await?;
        sidecar.wait_exit().await?;
        Ok(spec.output_path)
    }
    ```
