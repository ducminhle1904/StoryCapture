    ```rust
    pub fn resolve_font_path(choice: FontChoice) -> Result<PathBuf, EffectsError> {
        let rel = match choice {
            FontChoice::GeistSansRegular => "Geist-Regular.ttf",
            FontChoice::GeistSansBold => "Geist-Bold.ttf",
            FontChoice::JetBrainsMonoRegular => "JetBrainsMono-Regular.ttf",
            FontChoice::InterDisplay => "Inter-Display.ttf",
            FontChoice::SpaceGroteskDisplay => "SpaceGrotesk-Display.ttf",
        };
        let base = workspace_root().join("assets").join("fonts").join(rel);
        if !base.exists() { return Err(EffectsError::FontMissing(rel.to_string())); }
        Ok(base)
    }
    pub fn ensure_fonts_extracted(into: &Path) -> Result<PathBuf, EffectsError> {
        // Pitfall #8: if `into` contains spaces, copy fonts into a UUID-named subdir that is known space-free.
        let target = into.join(format!("storycapture_fonts_{}", uuid::Uuid::new_v4().simple()));
        std::fs::create_dir_all(&target)?;
        for f in ["Geist-Regular.ttf","Geist-Bold.ttf","JetBrainsMono-Regular.ttf","Inter-Display.ttf","SpaceGrotesk-Display.ttf"] {
            std::fs::copy(resolve_font_path_by_name(f)?, target.join(f))?;
        }
        Ok(target)
    }
    pub fn path_to_ffmpeg_arg(p: &Path) -> String {
        // Always forward slashes; assume `ensure_fonts_extracted` has removed spaces.
        p.to_string_lossy().replace('\\', "/")
    }
    ```

    ```rust
    pub fn escape_drawtext_text(s: &str) -> String {
        // FFmpeg filter arg escaping: `:`, `\`, `'`, `%` and the special inline sequences.
        s.replace('\\', "\\\\").replace(':', "\\:").replace('\'', "\\'").replace('%', "\\%")
    }
    pub fn emit_drawtext(tb: &TextBox, font_dir: &Path, in_label: &str, out_label: &str) -> Result<String, EffectsError> {
        let font_path = path_to_ffmpeg_arg(&font_dir.join(font_filename(tb.font)));
        let alpha_expr = crate::text::animation::compose_alpha_expr(tb);
        let text = escape_drawtext_text(&tb.text);
        let enable = format!("between(t,{},{})", tb.t_start_ms as f64 / 1000.0, tb.t_end_ms as f64 / 1000.0);
        Ok(format!(
            "{in_label}drawtext=fontfile='{font}':text='{text}':x={x}:y={y}:fontcolor=0x{fg:08X}:fontsize={size}:alpha='{alpha}':enable='{enable}'{out_label}",
            in_label = in_label, font = font_path, text = text, x = tb.pos.x as i32, y = tb.pos.y as i32,
            fg = rgba_to_hex(&tb.color), size = tb.size_pt as u32, alpha = alpha_expr, enable = enable, out_label = out_label
        ))
    }
    ```

    ```rust
    pub fn anim_fade_params(t_start_ms: u64, t_end_ms: u64, in_ms: u64, out_ms: u64) -> String {
        let t0 = t_start_ms as f64 / 1000.0;
        let t1 = (t_start_ms + in_ms) as f64 / 1000.0;
        let t2 = (t_end_ms - out_ms) as f64 / 1000.0;
        let t3 = t_end_ms as f64 / 1000.0;
        format!("if(lt(t,{t0}),0,if(lt(t,{t1}),(t-{t0})/{in_s},if(lt(t,{t2}),1,if(lt(t,{t3}),1-(t-{t2})/{out_s},0))))",
            t0 = t0, t1 = t1, t2 = t2, t3 = t3, in_s = in_ms as f64 / 1000.0, out_s = out_ms as f64 / 1000.0)
    }
    pub fn compose_alpha_expr(tb: &TextBox) -> String {
        let in_ms = match tb.anim_in { TextAnim::None => 0, _ => 300 };
        let out_ms = match tb.anim_out { TextAnim::None => 0, _ => 300 };
        anim_fade_params(tb.t_start_ms, tb.t_end_ms, in_ms, out_ms)
    }
    // slide-up + scale-in require drawtext's x/y expressions too; emit_drawtext reads them.
    ```
