    pub struct BatchExportRequest { pub outputs: Vec<(OutputFormat, Resolution, u32 /* fps */, Quality)>, pub out_folder: PathBuf, pub base_name: String }
    pub struct OutputSpec { pub id: Uuid, pub batch_id: Uuid, pub format: OutputFormat, pub resolution: Resolution, pub fps: u32, pub quality: Quality, pub output_path: PathBuf }
    pub fn build_batch(req: &BatchExportRequest) -> Result<Vec<OutputSpec>, ExportError> {
        let batch_id = Uuid::now_v7();
        req.outputs.iter().map(|(fmt, res, fps, q)| {
            validate(*fmt, *res, *fps)?;
            let filename = format!("{}.{}.{}.{}", req.base_name, res_label(*res), fps, fmt.extension());
            Ok(OutputSpec { id: Uuid::new_v4(), batch_id, format: *fmt, resolution: *res, fps: *fps, quality: *q, output_path: req.out_folder.join(filename) })
        }).collect()
    }
    fn validate(fmt: OutputFormat, res: Resolution, fps: u32) -> Result<(), ExportError> {
        if !VALID_FPS.contains(&fps) { return Err(ExportError::InvalidFps(fps)); }
        if matches!(fmt, OutputFormat::Gif) && matches!(res, Resolution::R4k) { return Err(ExportError::UnsupportedCombination); }
        if matches!(fmt, OutputFormat::Gif) && fps > 30 { return Err(ExportError::UnsupportedCombination); }
        Ok(())
    }

    pub struct PsnrResult { pub average: f64, pub min: f64, pub max: f64 }
    pub async fn compute_psnr(ffmpeg_path: &Path, reference: &Path, candidate: &Path) -> Result<PsnrResult, EncoderError> {
        // ffmpeg -i candidate.mp4 -i reference.mp4 -lavfi "[0:v][1:v]psnr=stats_file=-" -f null -
        let out = tokio::process::Command::new(ffmpeg_path)
            .arg("-hide_banner").arg("-i").arg(candidate).arg("-i").arg(reference)
            .arg("-lavfi").arg("[0:v][1:v]psnr=stats_file=-")
            .arg("-f").arg("null").arg("-")
            .output().await?;
        parse_psnr_stats(&String::from_utf8_lossy(&out.stderr))
    }
    fn parse_psnr_stats(stderr: &str) -> Result<PsnrResult, EncoderError> {
        // Look for "Parsed_psnr_0 ... average:40.12 ..." line or parse individual frame lines.
        // FFmpeg prints "PSNR y:40.28 u:44.21 v:44.29 average:41.71 min:32.14 max:49.88"
        let line = stderr.lines().rev().find(|l| l.contains("average:")).ok_or(EncoderError::PsnrParse)?;
        let avg = extract_num(line, "average:")?;
        let min = extract_num(line, "min:")?;
        let max = extract_num(line, "max:")?;
        Ok(PsnrResult { average: avg, min, max })
