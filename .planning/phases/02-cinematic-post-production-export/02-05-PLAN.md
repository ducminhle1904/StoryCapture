Implement the auto-zoom planner (POST-02) that converts Phase 1 click/hover/scroll waypoints into smooth pan-and-scale keyframes following D-06 discipline (pan first → scale in → hold, never combine). Ship three presets (Dynamic default per D-05, Calm, Subtle pan-only), the critically-damped-spring low-pass smoothing (omega=6 per Research §4), and wire the output into `VideoNode::ZoomPan`'s filter_complex emission so both the WebGPU preview and FFmpeg final renderer consume the same smoothed keyframes.

    impl<'a> WaypointSource for SqliteWaypointSource<'a> {
        fn load(&self, story_id: Uuid) -> Result<Vec<Waypoint>, EffectsError> {
            let mut stmt = self.conn.prepare("SELECT t_ms, x, y, kind FROM steps WHERE story_id = ?1 ORDER BY t_ms ASC")?;
            let rows = stmt.query_map([story_id.to_string()], |row| {

                let kind = match kind_str.as_str() {
                    "click" => WaypointKind::Click, "hover" => WaypointKind::Hover, "scroll" => WaypointKind::Scroll,
                    "type" => WaypointKind::Type, "drag" => WaypointKind::Drag,
                    other => return Err(rusqlite::Error::FromSqlConversionFailure(3, rusqlite::types::Type::Text, Box::new(EffectsError::UnknownWaypointKind(other.to_string())))),

                Ok(Waypoint { t_ms: row.get::<_, i64>(0)? as u64, pos: Vec2 { x: row.get::<_, f64>(1)? as f32, y: row.get::<_, f64>(2)? as f32 }, kind })
            })?;
            rows.collect::<Result<Vec<_>, _>>().map_err(Into::into)
        }
    }

    ```rust
    pub fn build_keyframes(clusters: &[ZoomCluster], preset: &ZoomPreset, viewport_w: u32, viewport_h: u32) -> Vec<ZoomKeyframe> {
        let mut out = Vec::new();
        let mut prev_center = Vec2 { x: viewport_w as f32 / 2.0, y: viewport_h as f32 / 2.0 };
        let mut prev_scale: f32 = 1.0;
        let mut t = 0u64;
        // Initial hold at identity
        out.push(ZoomKeyframe { t_ms: 0, center: prev_center, scale: 1.0, easing: EasingKind::Linear });
        for cluster in clusters {
            let pan_end = t.max(cluster.t_start_ms).saturating_sub(preset.pan_duration_ms as u64 + preset.scale_duration_ms as u64);
            // Phase 1: PAN from prev_center -> cluster.center at prev_scale
            out.push(ZoomKeyframe { t_ms: pan_end, center: prev_center, scale: prev_scale, easing: EasingKind::EaseInOutCubic });
            let after_pan = pan_end + preset.pan_duration_ms as u64;
            out.push(ZoomKeyframe { t_ms: after_pan, center: cluster.center, scale: prev_scale, easing: EasingKind::EaseInOutCubic });
            // Phase 2: SCALE from prev_scale -> cluster.scale (skip if pan_only)
            let target_scale = if preset.pan_only { prev_scale } else { cluster.scale.min(preset.max_zoom) };
            if !preset.pan_only {
                let after_scale = after_pan + preset.scale_duration_ms as u64;
                out.push(ZoomKeyframe { t_ms: after_scale, center: cluster.center, scale: target_scale, easing: EasingKind::EaseInOutCubic });
                t = after_scale;
            } else {
                t = after_pan;
            }
            // Phase 3: HOLD until cluster.t_end_ms
            out.push(ZoomKeyframe { t_ms: cluster.t_end_ms, center: cluster.center, scale: target_scale, easing: EasingKind::Linear });
            prev_center = cluster.center;
            prev_scale = target_scale;
            t = cluster.t_end_ms;
        }
        out
    }
    ```
