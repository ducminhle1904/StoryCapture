       ```rust
       //! Phase 10 author-time simulator — Tauri command surface.
       //!
       //! Internal name is "simulator" (D-00 in 10-CONTEXT.md) to avoid collision
       //! with Phase 3's shipped `intelligence::dryrun` / `DryRunPanel.tsx` /

       //!
       //! This module is a PURE Tauri-command wiring layer. It consumes the public
       //! automation crate APIs shipped by 10-01 Task 4: `run_story`, `continue_run`,
       //! `RunControl::cancel`, and `try_promote_fallback`. It does NOT modify any
       //! files under crates/automation/.

       use std::collections::HashMap;
       use std::path::{Path, PathBuf};
       use std::sync::Arc;
       use std::time::SystemTime;

       use serde::{Deserialize, Serialize};
       use tauri::ipc::Channel;
       use tauri::{AppHandle, Manager, State};
       use tokio::sync::Mutex;
       use tokio::task::JoinHandle;

       use automation::{ExecutorEvent, RunControl, StepFrame};

       use crate::error::AppError; // same AppError type used by commands/automation.rs

       pub type SimulatorSessionId = String; // UUID v4 string

       /// Resumable run state — held between `simulator_step_to` calls so
       /// subsequent invocations do NOT relaunch Chromium (Pitfall 1).
       pub struct ResumableSession {
           pub stream_id: String,        // from 09-04 start_author_preview
           pub project_folder: PathBuf,
           pub story_path: PathBuf,
           pub run_id: String,           // UUID v4 — dir name under .story.simulator/
           pub last_ordinal: u32,        // advanced on RunPaused/StepSucceeded; read by continue_run
           pub control: Arc<RunControl>,
           pub task: Option<JoinHandle<()>>,
           /// Snapshot of captured frames by ordinal — used by simulator_promote_fallback
           /// to look up the matched_selector for a step.
           pub frames: HashMap<u32, StepFrame>,
           /// Drivers are held here so continue_run can reuse them (no relaunch).
           pub primary: Option<Box<dyn automation::BrowserDriver>>,
           pub fallback: Option<Box<dyn automation::BrowserDriver>>,
       }

       #[derive(Default)]
       pub struct SimulatorRegistry {
           pub sessions: Mutex<HashMap<SimulatorSessionId, ResumableSession>>,
       }

       #[derive(Debug, Clone, Serialize, Deserialize)]
       #[serde(tag = "type", rename_all = "snake_case")]
       pub enum SimulatorEvent {
           Started { session_id: String, run_id: String, total_steps: u32 },
           FrameCaptured { ordinal: u32, frame: StepFrame },
           Paused { ordinal: u32 },
           Failed { ordinal: u32, error_message: String },
           Completed { succeeded: u32, failed: u32 },
           Cancelled,
       }

       #[derive(Debug, thiserror::Error, Serialize)]
       pub enum SimulatorError {
           #[error("preview is disabled — enable Preview in the Editor header before running the simulator")]
           PreviewDisabled,
           #[error("ordinal {ordinal} out of range (story has {total} steps)")]
           OrdinalOutOfRange { ordinal: u32, total: u32 },
           #[error("session {0} not found — call simulator_start first")]
           SessionNotFound(String),
           #[error("author preview not attached: {0}")]
           AuthorAttach(String),
           #[error("io: {0}")]
           Io(String),
       }

       /// Retention trim — keep the 5 most recent run dirs under
       /// `<project_folder>/.story.simulator/`. Returns number of dirs deleted.
       pub fn prune_runs_retain_5(project_folder: &Path) -> std::io::Result<usize> {
           let root = project_folder.join(".story.simulator");
           if !root.exists() {
               std::fs::create_dir_all(&root)?;
               return Ok(0);
           }
           let mut entries: Vec<(PathBuf, SystemTime)> = std::fs::read_dir(&root)?
               .filter_map(|e| e.ok())
               .filter(|e| e.file_type().map(|t| t.is_dir()).unwrap_or(false))
               .filter_map(|e| {
                   let mtime = e.metadata().and_then(|m| m.modified()).ok()?;
                   Some((e.path(), mtime))
               })
               .collect();
           entries.sort_by(|a, b| b.1.cmp(&a.1)); // newest first
           let mut deleted = 0;
           for (path, _) in entries.into_iter().skip(5) {
               let _ = std::fs::remove_dir_all(&path);
               deleted += 1;
           }
           Ok(deleted)
       }
       ```

       ```rust
       use std::time::{Duration, SystemTime};
       use tempfile::tempdir;

       use storycapture_desktop::prune_runs_retain_5;

       fn touch(dir: &std::path::Path, age_secs: u64) {
           fs::create_dir_all(dir).unwrap();
           let t = SystemTime::now() - Duration::from_secs(age_secs);
           filetime::set_file_mtime(dir, filetime::FileTime::from_system_time(t)).unwrap();
       }

       #[test]
       fn retains_only_5_most_recent() {
           let project = tempdir().unwrap();
           let sim = project.path().join(".story.simulator");
           fs::create_dir_all(&sim).unwrap();
           for i in 0..7u64 {
               touch(&sim.join(format!("run-{i}")), i * 10);  // 0 = newest
           }
           let deleted = prune_runs_retain_5(project.path()).unwrap();
           assert_eq!(deleted, 2);
           let remaining: Vec<String> = fs::read_dir(&sim)
               .unwrap()
               .filter_map(|e| e.ok())
               .map(|e| e.file_name().to_string_lossy().into_owned())
               .collect();
           assert_eq!(remaining.len(), 5);
           assert!(!remaining.contains(&"run-5".to_string()));
           assert!(!remaining.contains(&"run-6".to_string()));
       }

       #[test]
       fn under_5_keeps_all() {
           let project = tempdir().unwrap();

           fs::create_dir_all(&sim).unwrap();
           for i in 0..3u64 {

           let deleted = prune_runs_retain_5(project.path()).unwrap();
       ```
