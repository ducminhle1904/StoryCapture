    export type QualityControlSpec =
      | { kind: "slider-crf"; min: 0; max: 51; default: number }
      | { kind: "slider-cq";  min: 0; max: 51; default: number }
      | { kind: "number-bitrate-mbps"; min: 1; max: 50; default: number }
      | { kind: "auto-hide"; note: string }
      | { kind: "none" };

    export interface QualityControlsResult {
      rateControlOptions: ReadonlyArray<{ value: ExportKnobs["rateControl"]; locked?: boolean }>;
      qualityControl: QualityControlSpec;
      presetOptions: ReadonlyArray<ExportKnobs["x264Preset"] | string>;
      note?: string;
    }

    export function deriveQualityControls(
      encoder: ExportKnobs["hwEncoder"],
      codec: ExportKnobs["codec"],
    ): QualityControlsResult {
      switch (encoder) {
        case "auto": return { rateControlOptions: [], qualityControl: { kind: "auto-hide", note: LABEL_AUTO_HIDE_NOTE }, presetOptions: [] };
        case "software":
        case "libx264": return {
          rateControlOptions: [{ value: "crf" }, { value: "cbr" }, { value: "vbr" }],
          qualityControl: { kind: "slider-crf", min: 0, max: 51, default: 18 },
          presetOptions: ["ultrafast","superfast","veryfast","faster","fast","medium","slow","slower","veryslow"],
        };
        case "h264-nvenc":
        case "hevc-nvenc": return {
          rateControlOptions: [{ value: "vbr", locked: true }],
          qualityControl: { kind: "slider-cq", min: 0, max: 51, default: 19 },
          presetOptions: ["p1","p2","p3","p4","p5","p6","p7"],
        };
        case "h264-videotoolbox":
        case "hevc-videotoolbox": return {
          rateControlOptions: [{ value: "vbr" }],
          qualityControl: { kind: "number-bitrate-mbps", min: 1, max: 50, default: 8 },
          presetOptions: ["speed","quality"],
        };
        case "h264-qsv": return {
          rateControlOptions: [{ value: "vbr" }, { value: "cbr" }],
          qualityControl: { kind: "number-bitrate-mbps", min: 1, max: 50, default: 8 },
          presetOptions: ["veryfast","faster","fast","medium","slow","slower"],
        };
        case "h264-amf": return {
          rateControlOptions: [{ value: "vbr" }, { value: "cbr" }],
          qualityControl: { kind: "number-bitrate-mbps", min: 1, max: 50, default: 8 },
          presetOptions: ["speed","balanced","quality"],
        };
        case "libopenh264": return {
          rateControlOptions: [{ value: "cbr" }],
          qualityControl: { kind: "number-bitrate-mbps", min: 1, max: 50, default: 4 },
          presetOptions: [],
          note: LABEL_LIBOPENH264_NOTE,
        };
        default: return {
          rateControlOptions: [{ value: "vbr" }],
          qualityControl: { kind: "number-bitrate-mbps", min: 1, max: 50, default: 8 },
          presetOptions: [],
        };
      }
    }
    ```

    /**
     * AdvancedOutputOptions — 8 export-only knobs grouped into 3 visual sub-groups.
     *
     * Phase 13 ENC-13 + ENC-16 + CD-13-01. Conditional fields per encoder via
     * deriveQualityControls(). HW encoder list = probe + Software fallback.
     */
    import { useQuery } from "@tanstack/react-query";
    import { useOutputPrefsStore } from "@/state/output-prefs";
    import { probeHwEncoders } from "@/ipc/encode";
    import { deriveQualityControls } from "./encoder-options-table";
    import * as copy from "./advanced-copy";
    // ... primitives from @/components/ui/*

    export function AdvancedOutputOptions() {
      const exportKnobs = useOutputPrefsStore((s) => s.exportKnobs);
      const setExportKnob = useOutputPrefsStore((s) => s.setExportKnob);

      const { data: probeRaw, isLoading: probing } = useQuery({
        queryKey: ["hw-encoders"],
        queryFn: probeHwEncoders,
        staleTime: Infinity,
        refetchOnMount: false,
        refetchOnWindowFocus: false,
      });
      const availableEncoders = useMemo(() => parseProbe(probeRaw), [probeRaw]);
      const hwUnavailable = exportKnobs.hwEncoder !== "auto"
        && exportKnobs.hwEncoder !== "software"
        && !availableEncoders.find((e) => e.kind === exportKnobs.hwEncoder);

      const quality = deriveQualityControls(exportKnobs.hwEncoder, exportKnobs.codec);

      return (
        <div className="grid grid-cols-1 gap-4 md:grid-cols-2 md:gap-x-6 md:gap-y-4">
          {/* Group 1 — Container & Codec */}
          <SubGroup label="Container & Codec">
            <Field label={copy.LABEL_CONTAINER}>
              <Select value={exportKnobs.container} onChange={(v) => setExportKnob("container", v)}>
                <option value="mp4">MP4</option>
                <option value="mov">MOV</option>
                <option value="web-m" disabled>WebM (chưa hỗ trợ)</option>
              </Select>
            </Field>
            <Field label={copy.LABEL_CODEC}>
              <Select value={exportKnobs.codec} onChange={(v) => setExportKnob("codec", v)}>
                <option value="h264">H.264</option>
              </Select>
            </Field>
          </SubGroup>

          {/* Group 2 — Encoder & Quality (conditional) */}
          <SubGroup label="Encoder & Quality">
            <Field label={copy.LABEL_HW_ENCODER}>
              <Select value={exportKnobs.hwEncoder} onChange={(v) => setExportKnob("hwEncoder", v)}>
                <option value="auto">Auto</option>
                {availableEncoders.map((e) => <option key={e.kind} value={e.kind}>{labelFor(e.kind)}</option>)}
                <option value="software">Software (libx264)</option>
                {hwUnavailable && (
                  <option value={exportKnobs.hwEncoder} disabled>
                    {exportKnobs.hwEncoder} (không khả dụng trên máy này)
                  </option>
                )}
              </Select>
            </Field>
            {hwUnavailable && (
              <div role="status" aria-live="polite" className="text-xs text-[var(--color-warning)]">
                {copy.WARN_HW_UNAVAILABLE(exportKnobs.hwEncoder)}
              </div>
            )}
            {quality.qualityControl.kind === "auto-hide" ? (
              <div className="text-xs text-[var(--color-fg-muted)]">{quality.qualityControl.note}</div>
            ) : (
              <>
                {quality.rateControlOptions.length > 0 && (
                  <Field label={copy.LABEL_RATE_CONTROL}>
                    <RadioGroup value={exportKnobs.rateControl} onChange={(v) => setExportKnob("rateControl", v)}>
                      {quality.rateControlOptions.map((opt) => (
                        <Radio key={opt.value} value={opt.value} disabled={opt.locked}>{labelForRC(opt.value)}</Radio>
                      ))}
                    </RadioGroup>
                  </Field>
                )}
                {quality.qualityControl.kind === "slider-crf" || quality.qualityControl.kind === "slider-cq" ? (
                  <Field label={copy.LABEL_QUALITY_SLIDER}>
                    <Slider min={quality.qualityControl.min} max={quality.qualityControl.max}
                            value={exportKnobs.qualityValue ?? quality.qualityControl.default}
                            onChange={(n) => setExportKnob("qualityValue", n)} />
                  </Field>
                ) : quality.qualityControl.kind === "number-bitrate-mbps" ? (
                  <Field label={copy.LABEL_BITRATE_MBPS}>
                    <NumberField min={quality.qualityControl.min} max={quality.qualityControl.max}
                                 value={exportKnobs.qualityValue ?? quality.qualityControl.default}
                                 onChange={(n) => setExportKnob("qualityValue", typeof n === "number" ? n : null)} />
                  </Field>
                ) : null}
                {quality.presetOptions.length > 0 && (
                  <Field label={copy.LABEL_PRESET}>
                    <Select value={exportKnobs.x264Preset} onChange={(v) => setExportKnob("x264Preset", v)}>
                      {quality.presetOptions.map((p) => <option key={p} value={p}>{p}</option>)}
                    </Select>
                  </Field>
                )}
              </>
            )}
          </SubGroup>

          {/* Group 3 — Keyframe / Downscale / Audio */}
          <SubGroup label="Keyframe / Downscale / Audio" className="md:col-span-2">
            <Field label={copy.LABEL_KEYFRAME}>
              <NumberField min={1} max={10} value={exportKnobs.keyframeSec}
                           onChange={(n) => setExportKnob("keyframeSec", typeof n === "number" ? n : 2)} />
            </Field>
            <Field label={copy.LABEL_DOWNSCALE}>
              <RadioGroup value={exportKnobs.downscaleAlgo} onChange={(v) => setExportKnob("downscaleAlgo", v)}>
                <Radio value="lanczos">Lanczos</Radio>
                <Radio value="bicubic">Bicubic</Radio>
                <Radio value="bilinear">Bilinear</Radio>
              </RadioGroup>
            </Field>
            <Field label={copy.LABEL_AUDIO_CODEC}>
              <Select value={exportKnobs.audio.codec} onChange={(v) => setExportKnob("audio", { ...exportKnobs.audio, codec: v })}>
                <option value="aac">AAC</option>
                <option value="opus" disabled>Opus (WebM only — chưa hỗ trợ)</option>
              </Select>
            </Field>
            <Field label={copy.LABEL_AUDIO_BITRATE}>
              <Slider min={64} max={320} step={32} value={exportKnobs.audio.bitrateKbps}
                      onChange={(n) => setExportKnob("audio", { ...exportKnobs.audio, bitrateKbps: n })} />
            </Field>
            <Field label={copy.LABEL_AUDIO_CHANNELS}>
              <RadioGroup value={String(exportKnobs.audio.channels)}
                          onChange={(v) => setExportKnob("audio", { ...exportKnobs.audio, channels: Number(v) as 1 | 2 })}>
                <Radio value="1">Mono</Radio>
                <Radio value="2">Stereo</Radio>
              </RadioGroup>
            </Field>
          </SubGroup>
        </div>
      );
    }
    ```
