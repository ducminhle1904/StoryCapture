    ```ts
     * Shared output-prefs store. Cross-feature: consumed by the Recording View
     * (5 knobs) AND the Export Modal (8 export-only knobs on top).
     *
     * Phase 13 — second documented slice-composition exception (see docs/CONVENTIONS.md).

    import type {
      OutputResolutionDto,
      FitModeDto,
      PadColorDto,
      QualityPresetDto,
      ScaleAlgoDto,
    } from "@storycapture/shared-types/ipc";

    export type PresetName = "Quick" | "Standard" | "High Quality" | "Custom";

    export interface RecordingKnobs {
      resolution: OutputResolutionDto;
      fps: number;
      fit: FitModeDto;
      pad: PadColorDto;
      quality: QualityPresetDto;
    }

    export interface AudioKnobs {
      codec: "aac" | "opus";
      bitrateKbps: number;
      channels: 1 | 2;
      sampleRateHz: number;
    }

    export interface ExportKnobs {
      container: "mp4" | "mov" | "web-m";
      codec: "h264";
      rateControl: "auto" | "cbr" | "vbr" | "crf" | "cq";
      hwEncoder: string; // "auto" | "software" | probed name
      x264Preset:
        | "ultrafast" | "superfast" | "veryfast" | "faster" | "fast"
        | "medium" | "slow" | "slower" | "veryslow";
      keyframeSec: number;
      downscaleAlgo: "lanczos" | "bicubic" | "bilinear";
      audio: AudioKnobs;
      /** Single nullable numeric whose meaning is discriminated by
       *  (codec, hwEncoder, rateControl) via Plan 13-05's deriveQualityControls():
       *    - libx264 + crf → CRF integer 0..=51
       *    - nvenc + vbr   → CQ  integer 0..=51
       *    - videotoolbox/qsv/amf → bitrate Mbps
       *    - auto / libopenh264 → null (backend default)
       *  Null = use Phase 12 pixel_based default (CD-13-04 anchor). */
      qualityValue: number | null;
    }

    export const PRESET_BUNDLES: Record<Exclude<PresetName, "Custom">, RecordingKnobs> = {
      Quick:          { resolution: { kind: "P720"  }, fps: 30, fit: "letterbox", pad: { kind: "black" }, quality: "low"  },
      Standard:       { resolution: { kind: "P1080" }, fps: 30, fit: "letterbox", pad: { kind: "black" }, quality: "med"  },
      "High Quality": { resolution: { kind: "P1080" }, fps: 60, fit: "letterbox", pad: { kind: "black" }, quality: "high" },
    };

    export const DEFAULT_EXPORT_KNOBS: ExportKnobs = {
      container: "mp4",
      codec: "h264",
      rateControl: "auto",
      hwEncoder: "auto",
      x264Preset: "medium",
      keyframeSec: 2,
      downscaleAlgo: "lanczos",
      audio: { codec: "aac", bitrateKbps: 160, channels: 2, sampleRateHz: 48_000 },
      // Phase 12 pixel_based fallback per CD-13-04 — anchored to current backend behaviour.
      qualityValue: null,
    };

    export function matchPreset(knobs: RecordingKnobs): Exclude<PresetName, "Custom"> | null {
      for (const [name, bundle] of Object.entries(PRESET_BUNDLES)) {
        if (deepEqual(knobs, bundle)) return name as Exclude<PresetName, "Custom">;
      }
      return null;
    }

    function deepEqual(a: unknown, b: unknown): boolean {
      // Strict structural equality for the tiny knob shape — no need to pull a lib.
      return JSON.stringify(a) === JSON.stringify(b);
    }

    interface State {
      activePreset: PresetName;
      recordingKnobs: RecordingKnobs;
      exportKnobs: ExportKnobs;
      setRecordingKnob<K extends keyof RecordingKnobs>(k: K, v: RecordingKnobs[K]): void;
      setExportKnob<K extends keyof ExportKnobs>(k: K, v: ExportKnobs[K]): void;
      applyPreset(name: Exclude<PresetName, "Custom">): void;
      hydrate(s: { activePreset: PresetName; recordingKnobs: RecordingKnobs; exportKnobs: ExportKnobs }): void;
    }

    export const useOutputPrefsStore = create<State>((set) => ({
      activePreset: "Standard",
      recordingKnobs: PRESET_BUNDLES.Standard,
      exportKnobs: DEFAULT_EXPORT_KNOBS,
      setRecordingKnob: (k, v) =>
        set((s) => {
          const next = { ...s.recordingKnobs, [k]: v };
          const matched = matchPreset(next);
          return { recordingKnobs: next, activePreset: matched ?? "Custom" };
        }),
      setExportKnob: (k, v) =>
        set((s) => ({ exportKnobs: { ...s.exportKnobs, [k]: v } })),
      applyPreset: (name) =>
        set({ activePreset: name, recordingKnobs: PRESET_BUNDLES[name] }),
      hydrate: ({ activePreset, recordingKnobs, exportKnobs }) =>
        set({ activePreset, recordingKnobs, exportKnobs }),
    }));
    ```

    ```ts
    /**
     * Output-prefs persistence — global (plugin-store) + per-project (.storycapture/output.json) IO + migrator.
     *
     * Phase 13: silent seed on first launch (D-13-06), debounced write-back on store changes,
     * precedence project > global > seed (D-13-05).
     */
    import { exists, readTextFile, writeTextFile, mkdir } from "@tauri-apps/plugin-fs";
    import { toast } from "sonner";
    import { getStore, STORE_KEY, LATEST_VERSION } from "@/ipc/output-prefs";
    import {
      DEFAULT_EXPORT_KNOBS,
      PRESET_BUNDLES,
      useOutputPrefsStore,
      type ExportKnobs,
      type PresetName,
      type RecordingKnobs,
    } from "@/state/output-prefs";

    export interface PersistShape {
      activePreset: PresetName;
      recordingKnobs: RecordingKnobs;
      exportKnobs: ExportKnobs;
      version: typeof LATEST_VERSION;
    }

    const SEED: PersistShape = {
      activePreset: "Standard",
      recordingKnobs: PRESET_BUNDLES.Standard,
      exportKnobs: DEFAULT_EXPORT_KNOBS,
      version: LATEST_VERSION,
    };

    export function migrate(raw: unknown): PersistShape {
      if (!raw || typeof raw !== "object") return SEED;
      const r = raw as Partial<PersistShape>;
      // Fill missing top-level fields from seed; deep-merge knob bags.
      return {
        activePreset: (r.activePreset ?? SEED.activePreset) as PresetName,
        recordingKnobs: { ...SEED.recordingKnobs, ...(r.recordingKnobs ?? {}) },
        exportKnobs: {
          ...SEED.exportKnobs,
          ...(r.exportKnobs ?? {}),
          audio: { ...SEED.exportKnobs.audio, ...((r.exportKnobs as ExportKnobs | undefined)?.audio ?? {}) },
          // qualityValue: spread above propagates user value; if absent → SEED's null.
        },
        version: LATEST_VERSION,
      };
    }

    export function resolveOverride(
      global: PersistShape,
      project: Partial<Omit<PersistShape, "version">> | null,
    ): PersistShape {
      if (!project) return global;
      return {
        activePreset: project.activePreset ?? global.activePreset,
        recordingKnobs: { ...global.recordingKnobs, ...(project.recordingKnobs ?? {}) },
        exportKnobs: {
          ...global.exportKnobs,
          ...(project.exportKnobs ?? {}),
          audio: { ...global.exportKnobs.audio, ...(project.exportKnobs?.audio ?? {}) },
        },
        version: LATEST_VERSION,
      };
    }

    export async function initOutputPrefs(): Promise<void> {
      try {
        const store = await getStore();
        const raw = await store.get<PersistShape>(STORE_KEY);
        const hydrated = migrate(raw);
        if (!raw || raw.version !== LATEST_VERSION) {
          // Silent seed (D-13-06) — no toast, no modal.
          await store.set(STORE_KEY, hydrated);
          await store.save();
        }
        useOutputPrefsStore.getState().hydrate(hydrated);
      } catch {
        // Storage unavailable — keep in-memory defaults (matches theme.ts pragma).
      }

      let t: ReturnType<typeof setTimeout> | undefined;
      useOutputPrefsStore.subscribe((s) => {
        if (t) clearTimeout(t);
        t = setTimeout(async () => {
          try {
            const store = await getStore();
            await store.set(STORE_KEY, {
              activePreset: s.activePreset,
              recordingKnobs: s.recordingKnobs,
              exportKnobs: s.exportKnobs,
              version: LATEST_VERSION,
            } satisfies PersistShape);
            await store.save();
          } catch {
            // swallow — write-back is best-effort
          }
        }, 250);
      });
    }

    const PROJECT_FILE_REL = ".storycapture/output.json";

    export async function loadProjectOverride(
      projectFolder: string,
    ): Promise<Partial<Omit<PersistShape, "version">> | null> {
      const path = `${projectFolder}/${PROJECT_FILE_REL}`;
      try {
        if (!(await exists(path))) return null;
        const text = await readTextFile(path);
        return JSON.parse(text);
      } catch {
        toast("Không đọc được tùy chọn riêng của dự án. Đang dùng mặc định chung.");
        return null;
      }
    }

    export async function saveProjectOverride(
      projectFolder: string,
      prefs: Partial<Omit<PersistShape, "version">>,
    ): Promise<void> {
      const dir = `${projectFolder}/.storycapture`;
      const path = `${projectFolder}/${PROJECT_FILE_REL}`;
      try {
        await mkdir(dir, { recursive: true });
        await writeTextFile(path, JSON.stringify(prefs, null, 2));
      } catch (err) {
        toast.error("Không lưu được tùy chọn vào dự án.");
        throw err;
      }
    }
    ```
