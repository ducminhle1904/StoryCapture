```rust
use std::collections::HashMap;
use std::sync::Mutex;
use tokio::task::AbortHandle;

#[derive(Default)]
pub struct NlTaskRegistry {
    inner: Mutex<HashMap<String, AbortHandle>>,
}

impl NlTaskRegistry {
    pub fn insert(&self, id: String, handle: AbortHandle) { self.inner.lock().unwrap().insert(id, handle); }
    pub fn abort(&self, id: &str) -> bool {
        if let Some(h) = self.inner.lock().unwrap().remove(id) { h.abort(); true } else { false }
    }
    pub fn remove(&self, id: &str) { self.inner.lock().unwrap().remove(id); }
}
```
