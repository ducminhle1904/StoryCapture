```rust
pub const OPENAI_TTS_URL: &str = "https://api.openai.com/v1/audio/speech";
pub const BUILTIN_VOICES: &[&str] = &["alloy", "echo", "fable", "onyx", "nova", "shimmer"];

pub struct OpenAiTtsProvider { http: reqwest::Client, api_key: Redacted<String> }

#[async_trait::async_trait]
impl TtsProvider for OpenAiTtsProvider {
    async fn synthesize(&self, req: TtsRequest) -> Result<Bytes, TtsError> {
        if !BUILTIN_VOICES.contains(&req.voice_id.as_str()) {
            return Err(TtsError::VoiceNotFound(req.voice_id));
        }
        let body = serde_json::json!({
            "model": req.model,   // "tts-1" or "tts-1-hd"
            "input": req.text,
            "voice": req.voice_id,
            "response_format": "mp3",
            "speed": 1.0
        });
        let resp = self.http.post(OPENAI_TTS_URL)
            .header("Authorization", format!("Bearer {}", self.api_key.expose()))
            .json(&body).send().await?;
        match resp.status().as_u16() {
            200 => Ok(resp.bytes().await?),
            401 | 403 => Err(TtsError::AuthFailed),
            402 => Err(TtsError::QuotaExceeded),
            429 => Err(TtsError::RateLimited { retry_after_s: parse_retry_after(&resp) }),
            code => Err(TtsError::Provider(format!("HTTP {}", code))),
        }
    }

    async fn list_voices(&self) -> Result<Vec<VoiceInfo>, TtsError> {
        Ok(BUILTIN_VOICES.iter().map(|v| VoiceInfo {
            id: v.to_string(), name: capitalize(v), locale: Some("en".into()), premium: false
        }).collect())
    }
}
```
