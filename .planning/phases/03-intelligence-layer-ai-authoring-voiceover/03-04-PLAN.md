```rust
pub fn backoff_delay(attempt: u32) -> std::time::Duration {
    let secs = (1u64 << attempt.min(5)).min(30);
    let jitter = rand::random::<u64>() % 1000;
    std::time::Duration::from_millis(secs * 1000 + jitter)
}

pub async fn with_backoff<F, Fut, T>(mut f: F) -> Result<T, super::LlmError>
where F: FnMut(u32) -> Fut, Fut: std::future::Future<Output = Result<T, super::LlmError>> {
    for attempt in 0..3u32 {
        match f(attempt).await {
            Ok(v) => return Ok(v),
            Err(super::LlmError::RateLimited { retry_after_s }) => {
                let wait = std::time::Duration::from_secs(retry_after_s).max(backoff_delay(attempt));
                tokio::time::sleep(wait).await;
            }
            Err(other) => return Err(other),
        }
    }
    Err(super::LlmError::Provider("retry exhausted".into()))
}
```
