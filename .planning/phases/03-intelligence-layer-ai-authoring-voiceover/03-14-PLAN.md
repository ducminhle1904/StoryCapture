```ts
import { invoke, Channel } from '@tauri-apps/api/core';

export function createTauriLspTransport(_docUri: string): TauriLspTransport {
  const channel = new Channel<{method: string; params: unknown}>();
  const handlers = new Set<(n: {method: string; params: unknown}) => void>();
  channel.onmessage = (n) => handlers.forEach(h => h(n));

  let nextId = 1;
  return {
    async sendRequest(method, params) {
      const envelope = { jsonrpc: "2.0", id: nextId++, method, params };
      const resp = await invoke<{result?: unknown; error?: {message: string}}>('lsp_request',
        { jsonrpcRequest: envelope, onNotification: channel });
      if (resp.error) throw new Error(resp.error.message);
      return resp.result;
    },
    sendNotification(method, params) {
      const envelope = { jsonrpc: "2.0", method, params };
      void invoke('lsp_request', { jsonrpcRequest: envelope, onNotification: channel });
    },
    onNotification(h) { handlers.add(h); return () => { handlers.delete(h); }; },
  };
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Webview → Tauri IPC (`lsp_request`) | Untrusted JSON-RPC envelopes |
| Rust → `LspService` | Should tolerate malformed requests |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-14-01 | Tampering | Malformed jsonrpc_request | mitigate | `jsonrpc::Request` parsing catches structural errors; invalid returns JSON-RPC error response, not panic |
| T-03-14-02 | Denial of Service | Flood of `lsp_request` calls | mitigate | DashMap operations are O(1); LspService backpressures via `ready().await`; Tauri command concurrency capped by runtime |
| T-03-14-03 | Information Disclosure | LSP hover/completion content | accept | Content derives from verb catalog + user-authored source; same data user already has |
| T-03-14-04 | Spoofing | Channel ID spoof | mitigate | `Channel<T>` is a Tauri primitive with per-window scoping; another webview cannot subscribe to someone else's channel |
</threat_model>
