pub fn hash_key(provider: &str, model: &str, voice_id: &str, script_text: &str) -> String {
    let mut h = sha2::Sha256::new();
    h.update(provider.as_bytes()); h.update(b"|");
    h.update(model.as_bytes()); h.update(b"|");
    h.update(voice_id.as_bytes()); h.update(b"|");
    h.update(script_text.as_bytes());
    hex::encode(h.finalize())
}

// Cache path (sanitized step_id)
pub fn cache_path(project_root: &Path, step_id: &str, hash: &str) -> PathBuf {
    project_root.join("voiceover").join(format!("{}-{}.mp3", sanitize_step_id(step_id), &hash[..16]))
}
