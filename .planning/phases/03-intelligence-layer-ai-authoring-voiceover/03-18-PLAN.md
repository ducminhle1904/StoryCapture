```ts
export function useDryRun(projectId: string) {
  const store = useDryRunStore();
  const start = useCallback(async (steps: StoryStep[]) => {
    const channel = new Channel<DryRunEvent>();
    channel.onmessage = (ev) => store.handleEvent(ev);
    const taskId = await invoke<string>('dryrun_start', { projectId, steps, channel });
    store.setTaskId(taskId);
  }, [projectId]);
  const cancel = useCallback(() => {
    if (store.taskId) invoke('dryrun_cancel', { taskId: store.taskId });
  }, []);
  return { start, cancel, state: store };
}
```
