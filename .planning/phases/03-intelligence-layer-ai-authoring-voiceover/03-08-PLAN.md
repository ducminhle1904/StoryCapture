```rust
#[async_trait::async_trait]
impl TtsProvider for ElevenLabsProvider {
    async fn synthesize(&self, req: TtsRequest) -> Result<Bytes, TtsError> {
        let url = format!("https://api.elevenlabs.io/v1/text-to-speech/{}/stream", req.voice_id);
        let body = serde_json::json!({
            "text": req.text,
            "model_id": req.model,  // "eleven_multilingual_v2"
            "voice_settings": {
                "stability":        req.stability.unwrap_or(0.5),
                "similarity_boost": req.similarity_boost.unwrap_or(0.75),
                "style":            0,
                "use_speaker_boost": true
            }
        });
        let resp = self.http.post(&url)
            .header("xi-api-key", self.api_key.expose())
            .header("Accept", "audio/mpeg")
            .json(&body)
            .send().await?;
        match resp.status().as_u16() {
            200 => { /* stream bytes into buffer */ }
            401 | 403 => return Err(TtsError::AuthFailed),
            402 => return Err(TtsError::QuotaExceeded),
            404 => return Err(TtsError::VoiceNotFound(req.voice_id)),
            429 => {
                let s = resp.headers().get("retry-after").and_then(|v| v.to_str().ok()).and_then(|s| s.parse::<u64>().ok()).unwrap_or(5);
                return Err(TtsError::RateLimited { retry_after_s: s });
            }
            code => return Err(TtsError::Provider(format!("HTTP {}", code))),
        }
        let mut buf = Vec::new();
        let mut stream = resp.bytes_stream();
        while let Some(chunk) = futures_util::StreamExt::next(&mut stream).await {
            buf.extend_from_slice(&chunk?);
        }
        Ok(Bytes::from(buf))
    }

    async fn list_voices(&self) -> Result<Vec<VoiceInfo>, TtsError> { /* GET /v1/voices, map .voices[] */ }
}
```
