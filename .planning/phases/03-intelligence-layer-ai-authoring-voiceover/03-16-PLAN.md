Implement `orchestrator.rs` `run` function:
```rust
pub async fn run<D: BrowserDriver + ?Sized>(
    driver: &D,
    steps: Vec<ExecStep>,
    task_id: String,
    tx: tokio::sync::mpsc::Sender<DryRunEvent>,
) -> Result<(), IntelError> {
    let total = steps.len() as u32;
    let mut passed = 0u32; let mut failed = 0u32; let mut total_ms = 0u64;
    for s in &steps {
        tx.send(DryRunEvent::Queued { step_id: s.id.clone() }).await.ok();
    }
    for s in steps {
        tx.send(DryRunEvent::Running { step_id: s.id.clone() }).await.ok();
        match driver.execute(&s).await {
            Ok(r) => {
                total_ms += r.elapsed_ms;
                passed += 1;
                tx.send(DryRunEvent::Pass { step_id: s.id, elapsed_ms: r.elapsed_ms, selector_attempts: r.selector_attempts }).await.ok();
            }
            Err(e) => {
                failed += 1;
                tx.send(DryRunEvent::Fail { step_id: s.id.clone(), error: e.to_string(), selector_attempts: e.selector_attempts() }).await.ok();
                break;  // stop on first failure per D-19 ("log selector fallback")
            }
        }
    }
    tx.send(DryRunEvent::Summary { total_steps: total, passed, failed, total_ms }).await.ok();
    tx.send(DryRunEvent::Done { task_id }).await.ok();
    Ok(())
}
```
