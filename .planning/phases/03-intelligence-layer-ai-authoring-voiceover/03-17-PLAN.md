// IPC types (generated by tauri-specta from Plan 07)
export type NlChatEvent =
  | { kind: "text"; delta: string }
  | { kind: "story_doc_ready"; doc: StoryDoc; diff: StepDiff[]; task_id: string }
  | { kind: "usage"; input: number; output: number; cache_read: number; cache_write: number; cost_usd: number }
  | { kind: "error"; message: string }
  | { kind: "done"; task_id: string };

// Zustand store shape
export interface NlStore {
  panelWidth: number;        // persisted via window.localStorage
  panelCollapsed: boolean;
  streaming: { taskId: string; text: string } | null;
  pendingCards: DiffCard[];  // per-step diff cards awaiting action
  error: { kind: "network" | "rate_limit" | "auth" ; message: string; retryAfterS?: number } | null;
  // actions
  setPanelWidth: (w: number) => void;
  togglePanel: () => void;
  beginStream: (taskId: string) => void;
  appendStream: (delta: string) => void;
  setCards: (cards: DiffCard[]) => void;
  clearCardsForTask: (taskId: string) => void;
  setError: (e: NlStore["error"]) => void;
}

export interface DiffCard {
  stepId: string;
  status: "pending" | "approved" | "rejected" | "regenerating";
  oldText?: string;
  newText?: string;
}
