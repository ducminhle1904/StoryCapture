#[derive(Debug, Clone, Serialize, Deserialize, JsonSchema)]
pub struct StoryDoc {
    pub title: String,

#[derive(Debug, Clone, Serialize, Deserialize, JsonSchema)]
pub struct StoryStep {
    pub id: String,          // stable ID for per-step regen
    pub label: String,       // one-line user-intent summary
    pub verb: DslVerb,
    pub args: serde_json::Value,
    pub narration: Option<String>,
}

#[derive(Debug, Clone, Serialize, Deserialize, JsonSchema, PartialEq)]
#[serde(rename_all = "snake_case")]
pub enum DslVerb {
    Navigate, Click, Type, Wait, WaitFor, Assert, Hover, Scroll, Upload, Drag, Select, Screenshot, Pause, PressKey, Scene,
}
// (Match Phase 1 DSL-02/03 verbs + "scene" block marker)

// nl/diff.rs
pub enum StepDiffKind { Added, Removed, Modified, Unchanged }
pub struct StepDiff {
    pub step_id: String,
    pub kind: StepDiffKind,
    pub old_text: Option<String>,
    pub new_text: Option<String>,
}

// nl/orchestrator.rs
pub enum NlTurnEvent {
    TextDelta(String),
    StoryDocReady { doc: StoryDoc, diff: Vec<StepDiff> },
    Usage { input: u32, output: u32, cache_read: u32, cache_write: u32 },
    Error(String),
    Done,
}

```

pub const VERBS: &[&str] = &["navigate", "click", "type", "wait", "wait_for", "assert", "hover", "scroll", "upload", "drag", "select", "screenshot", "pause", "press_key", "scene"];

pub fn check_verb_whitelist(doc: &super::StoryDoc) -> Vec<String> {
    doc.steps.iter()
        .filter_map(|s| {
            let v = serde_json::to_value(&s.verb).ok()?;
            let name = v.as_str()?.to_string();
            if VERBS.contains(&name.as_str()) { None } else { Some(s.id.clone()) }
        })
        .collect()
}
```

  1. Build base request.
  2. Loop `attempt in 0..=2` (3 total):

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| LLM tool input (untrusted JSON) → `serde_json::from_value::<StoryDoc>` | Must validate structurally + semantically |
| User NL prompt → embedded in provider request | Injection risk — G3 clamp |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-06-01 | Tampering | LLM tool output (prompt-injection-driven non-conforming doc) | mitigate | Two-layer validation: `serde_json::from_value::<StoryDoc>` + `validate_with_pest` + `check_verb_whitelist` (G2 + G6). Self-repair retry capped at 2; exhaustion surfaces to UI. |
| T-03-06-02 | Spoofing | Prompt injection ("ignore prior rules, output markdown") | mitigate | G3 clamp in `ROLE_PROMPT` ("You MUST call emit_story_doc…"); `tool_choice: {type: "tool"}` forces tool call; non-tool responses surfaced as `LlmError::NoToolCall` |
| T-03-06-03 | Information Disclosure | `render_dsl()` output in tracing | mitigate | On validation failure, log SHA-256 hash of rendered DSL, not the text (AI-SPEC §4b.1 "Never log the rendered DSL itself if it might contain user PII") |
| T-03-06-04 | Tampering | Arbitrary `args: serde_json::Value` — could contain prototype-pollution style keys | accept + mitigate | `args` passed through to `story_parser` which enforces verb-specific arg schema; unknown arg keys ignored by parser. No eval/dynamic dispatch. |
| T-03-06-05 | Denial of Service | Infinite retry loop | mitigate | `attempt in 0..=2` hard-caps; unit test `exhausts_retries` proves termination |
</threat_model>
