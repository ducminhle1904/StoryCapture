    SimulatorRunning { session: SimulatorSessionId },
    SimulatorPaused { session: SimulatorSessionId },

#[derive(Default)]
pub struct AuthorDriverRegistry {

#[derive(Debug, thiserror::Error, Serialize, specta::Type)]
#[serde(tag = "kind", content = "message")]
pub enum AuthorDriverError {
    #[error("Simulator running — cancel to pick")]
    SimulatorBusy,
    #[error("Pick already active")]
    AlreadyPicking,
    #[error("invalid transition from {from} to {to}")]
    InvalidTransition { from: String, to: String },
}

    pub fn begin_pick(&mut self, stream_id: StreamId); // boxes prior SimulatorPaused into resume_to
    pub fn end_pick(&mut self); // restore resume_to OR land in LivePreview{stream_id}
}

    Add three methods on `impl AuthorDriverState` (place immediately after `end_pick` at line 133):

    ```rust
    /// Transition into SimulatorRunning. Caller MUST have validated with
    /// can_start_simulator() first. Returns the prior state so the caller
    /// can stash it for end_simulator(). Mirror of begin_pick's snapshot-
    /// and-swap shape but without the resume_to box (simulator does not
    /// nest beneath Picking, so prior-state capture lives at the caller).
    pub fn begin_simulator(&mut self, session: SimulatorSessionId) -> AuthorDriverState {
        let prior = std::mem::replace(self, AuthorDriverState::SimulatorRunning { session });
        prior
    }

    /// Transition SimulatorRunning{s} -> SimulatorPaused{s}. No-op if state
    /// is not SimulatorRunning — guards against racey RunPaused events
    /// arriving after simulator_cancel has already restored the prior state.
    pub fn pause_simulator(&mut self) {
        let current = std::mem::replace(self, AuthorDriverState::Idle);
        *self = match current {
            AuthorDriverState::SimulatorRunning { session } => {
                AuthorDriverState::SimulatorPaused { session }
            }
            other => other, // no-op — restore as-was
        };
    }

    /// Exit simulator — restore `prior` captured by begin_simulator. No-op
    /// if current state is not Simulator* (guards against double-cancel
    /// where end_simulator runs twice — forwarder sees StoryEnded AND
    /// simulator_cancel races the forwarder).
    pub fn end_simulator(&mut self, prior: AuthorDriverState) {
        match self {
            AuthorDriverState::SimulatorRunning { .. } | AuthorDriverState::SimulatorPaused { .. } => {
                *self = prior;
            }
            _ => {} // already restored — no-op
        }
    }
    ```

| Boundary | Description |
|----------|-------------|
| renderer → Tauri command | `simulator_start` / `simulator_cancel` invocations cross from untrusted renderer into host; any param could be forged by a compromised renderer. |
