        driver: &automation::PlaywrightSidecarDriver,
        stream_id: &str,
        story_src: &str,
        cursor_line: u32,
    ) -> Result<(), AppError> {

                nav_urls.push(app_url);
            }

    Step 3 — Add `picker_start_author` Tauri command in `commands/picker.rs`. Follow 11-RESEARCH.md §Pattern 1 verbatim — lock-scope-drop-reacquire with PickerResumeGuard. Sketch:

    ```rust
    #[tauri::command]
    #[specta::specta]
    pub async fn picker_start_author(
        state: State<'_, AppState>,
        registry: State<'_, std::sync::Arc<crate::author_driver::AuthorDriverRegistry>>,
        stream_id: String,
        cursor_line: u32,
        timeout_ms: Option<u64>,
    ) -> Result<PickElementResponseDto, AppError> {
        let timeout = timeout_ms.unwrap_or(60_000);

        // 1. Scope: validate + transition + release lock.
        let prior_for_guard = {
            let mut g = registry.state.lock().await;
            g.can_start_pick().map_err(|e| AppError::Automation(e.to_string()))?;
            let prior = std::mem::replace(&mut *g, crate::author_driver::AuthorDriverState::Idle);
            let resume_to = matches!(&prior, crate::author_driver::AuthorDriverState::SimulatorPaused { .. })
                .then(|| Box::new(prior.clone()));
            *g = crate::author_driver::AuthorDriverState::Picking { stream_id: stream_id.clone(), resume_to };
            prior
        };

        // 2. RAII guard (11-01 PickerResumeGuard).
        let guard = crate::author_driver::PickerResumeGuard::arm(
            registry.inner().clone(), stream_id.clone(), prior_for_guard,
        );

        // 3. Acquire playwright driver (separate lock).
        let driver = {
            let slot = state.playwright_driver.lock().await;
            slot.as_ref().cloned().ok_or_else(|| AppError::Automation("Playwright sidecar not launched".into()))?
        };

        // 4. Navigate replay (best-effort). Reads from disk (AppState.open_story_path),
        //    NOT from the in-memory CM6 buffer (D-10 known constraint — see CONTEXT Deferred Ideas).
        //    Renderer warns the user via toast BEFORE invoking picker_start_author if the
        //    editor has unsaved changes (enforced in 11-04 Task 1; this host path trusts disk).
        let story_path = state.open_story_path.lock().await.clone()
            .ok_or_else(|| AppError::Automation("no open story".into()))?;
        let story_src = tokio::fs::read_to_string(&story_path).await
            .map_err(|e| AppError::Automation(format!("read {}: {e}", story_path.display())))?;
        let d = driver.lock().await;
        replay_navigate_verbs(&d, &stream_id, &story_src, cursor_line).await?;

        // 5. Pause preview (PHASE-9.9).
        drop(d);
        crate::commands::automation::pause_author_preview(stream_id.clone()).await?;

        // 6. Run picker.
        let d = driver.lock().await;
        let r = d.pick_element_start_author(&stream_id, timeout).await
            .map_err(|e| AppError::Automation(e.to_string()))?;
        drop(d);

        // 7. Resume preview (success path).
        crate::commands::automation::resume_author_preview(stream_id.clone()).await?;

        // 8. Disarm guard (success: state restore is what we want anyway — Drop does it idempotently).
        guard.disarm();

        // 9. Transition to LivePreview{stream_id} or restore SimulatorPaused per resume_to.
        {
            let mut g = registry.state.lock().await;
            g.end_pick();   // helper from 11-01
        }

        Ok(r.into())
    }
    ```

    Notes for implementers:
    - If `PickerResumeGuard::arm` is not the exact API 11-01 shipped, adapt — the key is construct-on-success-path-cleanup-on-Drop.
    - Navigate-replay reads from disk (`AppState.open_story_path`). The CM6 editor buffer is NOT plumbed host-side; using the on-disk bytes is the declared behavior per D-10 known constraint (CONTEXT Deferred Ideas — "in-memory open-story-source plumbing"). The renderer (11-04 Task 1) is responsible for warning the user via toast `Unsaved changes — Pick will use the last saved version. Save first?` when they click Pick while the editor is dirty. If they proceed, replay uses disk bytes.
    - `can_start_pick` + `end_pick` are 11-01 helpers. If 11-01 named them differently, align.
