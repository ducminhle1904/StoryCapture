      ```tsx
        if (picking) { pickElementCancel().catch(() => {}); return; }   // re-click = cancel
        const variant = useAuthorDriverStore.getState().variant;
        if (variant === "simulator-running") return;   // disabled; no-op
        const streamId = useAuthorDriverStore.getState().streamId;
        if (!streamId) { setIsStarting(true); /* trigger lazy-start via start_author_preview, await streamId */ }
        const cursorLine = editorController.getCursorLine();
        if (cursorLine == null) { toast.error("No cursor position"); return; }
        setPicking(true);
        try {
          const r = await pickElementAuthor({ streamId: streamId!, cursorLine, timeoutMs: 60_000 });
          if (isPicked(r)) {
            const res = editorController.insertAtCursor(r.emitted + "\n");
            if (res.ok) {
              // Per UI-SPEC §Toasts, distinguish first-pick vs re-pick — requires knowing whether the line was stamped.
              // For v1, call pickerStampStepId (idempotent) and rely on its "existing id" signal via a response flag or a pre-read.
              // Simplest: always use the first-pick copy when res.lineNumber corresponds to a freshly-stamped line;
              // fall back to a single generic "Inserted: {emitted}" if disambiguation is unclear.
              toast.success(`Added \`${r.emitted}\` · line ${res.lineNumber}`);
              const storyPath = editorController.getStoryPath();
              if (storyPath) {
                pickerStampStepId({ /* ... */ }).catch((e) => toast.error(/* ... */));
              }
            }
          } else {
            switch (r.reason) {
              case "user-cancel": break;                                 // silent per UI-SPEC
              case "navigation": toast.info("Picking cancelled — page navigated"); break;
              case "unsupported-url": toast.warning("Picking unavailable on this page (`chrome://`, `about:`, or `view-source:`)"); break;
              case "timeout": toast.error("Picker timed out — try again"); break;
            }
          }
        } catch (e) { toast.error(String(e)); }
        finally { setPicking(false); setIsStarting(false); }
      };
      ```
