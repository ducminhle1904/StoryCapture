    **`scripts/release/sign-windows.ps1`**:
    ```powershell
    [CmdletBinding()]
    param(
      [Parameter(Mandatory=$true)][string]$InstallerPath,
      [Parameter(Mandatory=$true)][ValidateSet("TrustedSigning", "EvCert")][string]$Mode
    )
    $ErrorActionPreference = "Stop"
    if ($Mode -eq "TrustedSigning") {
      # Microsoft Trusted Signing via azuresigntool or signtool with Trusted Signing extension
      $profile = $env:WINDOWS_SIGNING_CERT_NAME
      if (-not $profile) { throw "WINDOWS_SIGNING_CERT_NAME not set" }
      & azuresigntool sign -kvu $env:AZURE_KEY_VAULT_URL -kvt $env:AZURE_TENANT_ID -kvi $env:AZURE_CLIENT_ID -kvs $env:AZURE_CLIENT_SECRET -kvc $profile -tr "http://timestamp.digicert.com" -td sha256 -fd sha256 -v $InstallerPath
    } else {
      # EV cert path via Azure Key Vault
      $cert = $env:EV_CERT_AZURE_KV_NAME
      if (-not $cert) { throw "EV_CERT_AZURE_KV_NAME not set" }
      & azuresigntool sign -kvu $env:AZURE_KEY_VAULT_URL -kvt $env:AZURE_TENANT_ID -kvi $env:AZURE_CLIENT_ID -kvs $env:AZURE_CLIENT_SECRET -kvc $cert -tr "http://timestamp.digicert.com" -td sha256 -fd sha256 -v $InstallerPath
    }
    & signtool verify /pa /v $InstallerPath
    Write-Host "Signed: $InstallerPath"
    ```

    **`scripts/release/verify-installer-size.sh`**:
    ```bash
    #!/usr/bin/env bash
    set -euo pipefail
    INSTALLER="${1:?usage: $0 <installer-path> [--strict]}"
    STRICT="${2:-}"
    BUDGET_MB=50
    HARD_FAIL_MB=60

    TMP=$(mktemp -d)
    trap "rm -rf $TMP" EXIT

    case "$INSTALLER" in
      *.dmg)
        hdiutil attach -nobrowse -mountpoint "$TMP/mnt" "$INSTALLER"
        APP_DIR=$(find "$TMP/mnt" -maxdepth 2 -name "*.app" | head -1)
        total_bytes=$(du -s "$APP_DIR" | awk '{print $1 * 512}')
        ffmpeg_bytes=$(find "$APP_DIR" -name "ffmpeg-*" -exec du -k {} + | awk '{s+=$1} END {print s*1024}')
        sidecar_bytes=$(find "$APP_DIR" -name "playwright-sidecar-*" -exec du -k {} + | awk '{s+=$1} END {print s*1024}')
        hdiutil detach "$TMP/mnt"
        ;;
      *.exe|*.msi)
        7z x "$INSTALLER" -o"$TMP/ex" > /dev/null
        total_bytes=$(du -sb "$TMP/ex" | awk '{print $1}')
        ffmpeg_bytes=$(find "$TMP/ex" -name "ffmpeg*" -exec du -b {} + | awk '{s+=$1} END {print s+0}')
        sidecar_bytes=$(find "$TMP/ex" -name "playwright-sidecar*" -exec du -b {} + | awk '{s+=$1} END {print s+0}')
        ;;
      *) echo "Unknown format: $INSTALLER"; exit 1;;
    esac

    ex_ffmpeg=$((total_bytes - ffmpeg_bytes - sidecar_bytes))
    ex_ffmpeg_mb=$((ex_ffmpeg / 1024 / 1024))
    echo "Total: $((total_bytes/1024/1024)) MB  FFmpeg: $((ffmpeg_bytes/1024/1024)) MB  Playwright: $((sidecar_bytes/1024/1024)) MB  App residual: ${ex_ffmpeg_mb} MB"
    if [ "$ex_ffmpeg_mb" -gt "$HARD_FAIL_MB" ]; then
      echo "HARD FAIL: residual ${ex_ffmpeg_mb} MB > ${HARD_FAIL_MB} MB"
      exit 1
    fi
    if [ "$ex_ffmpeg_mb" -gt "$BUDGET_MB" ]; then
      if [ "$STRICT" = "--strict" ]; then
        echo "STRICT FAIL: residual ${ex_ffmpeg_mb} MB > ${BUDGET_MB} MB budget"
        exit 1
      else
        echo "WARN: residual ${ex_ffmpeg_mb} MB exceeds ${BUDGET_MB} MB budget (advisory)"
      fi
    fi
    echo "OK"
    ```

    ```yaml
    name: Installer Size Budget
    on:
      pull_request:
      push:

      size-check:
        strategy:

          matrix:
            include:

                target: aarch64-apple-darwin
                installer-glob: "**/*.dmg"
    ```

    ```yaml
    name: Release Soak
    on:
      workflow_run:
        workflows: ["Release"]
        types: [completed]
        branches-ignore: ["**"]  # only trigger on tag pushes; tag pushes have empty branch
      workflow_dispatch:
        inputs:
          tag_name:
            required: true
            type: string
    permissions:
      contents: read
    jobs:
      soak:
        if: github.event_name == 'workflow_dispatch' || github.event.workflow_run.conclusion == 'success'
        strategy:
          fail-fast: false
          matrix:
            os: [macos-14, windows-latest]
        runs-on: ${{ matrix.os }}
        timeout-minutes: 50
        steps:
          - uses: actions/checkout@v4
          - uses: ./.github/actions/setup-toolchain
          - name: Resolve tag
            id: tag
            shell: bash
            run: echo "tag=${{ github.event.inputs.tag_name || github.event.workflow_run.head_branch }}" >> $GITHUB_OUTPUT
          - name: Download release installer
            shell: bash
            run: |
              mkdir -p release-artifact
              gh release download "${{ steps.tag.outputs.tag }}" --dir release-artifact --pattern "*.dmg" --pattern "*-setup.exe"
            env:
              GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          - name: Install app
            shell: bash
            run: |
              if [ "${{ matrix.os }}" = "macos-14" ]; then
                dmg=$(ls release-artifact/*.dmg | head -1)
                hdiutil attach -nobrowse -mountpoint /tmp/mnt "$dmg"
                cp -R /tmp/mnt/*.app /Applications/
                hdiutil detach /tmp/mnt
                echo "APP_PATH=/Applications/StoryCapture.app" >> $GITHUB_ENV
              else
                exe=$(ls release-artifact/*-setup.exe | head -1)
                "$exe" /S
                echo "APP_PATH=C:/Program Files/StoryCapture/StoryCapture.exe" >> $GITHUB_ENV
              fi
          - name: Run soak against installed app
            shell: bash
            env:
              CAPTURE_SOAK_TARGET: ${{ env.APP_PATH }}
            run: cargo test -p capture --test soak --features real-capture -- --ignored --nocapture
          - uses: actions/upload-artifact@v4
            if: always()
            with:
              name: release-soak-${{ matrix.os }}-${{ steps.tag.outputs.tag }}
              path: target/tmp/rss-samples.csv
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GitHub Release → end user | Signed + notarized; Gatekeeper + SmartScreen verify offline. |
| Tauri updater private key → CI | Only in GitHub Secrets; never in repo, logs, or artifacts. |
| Azure Key Vault → Windows signing | Managed identity / service principal with least-privilege on the signing profile. |
| Updater public key → user install | Pinned in `tauri.conf.json`; rotation requires a new build (no remote key update). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-10-01 | Tampering | Malicious updater manifest | mitigate | Tauri updater enforces signature verification against pinned `pubkey`; mismatched signature aborts install. |
| T-10-02 | Spoofing | Phishing fake StoryCapture binary | mitigate | Code signing + notarization; users verify via `codesign` / SmartScreen. |
| T-10-03 | Information Disclosure | Private signing key leaked | mitigate | Keys stored only in GitHub Secrets; workflow logs masked; repo `.gitignore` excludes `.release-keys/`. |
| T-10-04 | Denial of Service | Auto-updater misconfigured → update loop | mitigate | `dialog: false` + user-controlled trigger; no automatic force-install. |
| T-10-05 | Elevation of Privilege | Installer sideloads arbitrary binary | mitigate | Tauri bundler only bundles declared `externalBin`; notarization verifies. |
| T-10-06 | Repudiation | Release fails signing but installer still published | mitigate | `release.yml` fails the job on signing error; no draft→published promotion if any matrix cell fails. |
| T-10-07 | Information Disclosure | Telemetry enabled by default | mitigate | Toggle defaults OFF; CI audit greps for new HTTP clients and warns. |
</threat_model>
