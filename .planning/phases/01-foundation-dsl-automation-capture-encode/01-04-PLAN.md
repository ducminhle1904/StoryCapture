<objective>
Build the Story DSL parser as a pure, Tauri-free library crate that parses `.story` files into a typed AST with span info on every node, uses a two-layer parse (lenient tokenize + semantic check) for human-readable diagnostics with Levenshtein "did you mean" suggestions and panic-mode recovery for multi-error reports. Auto-generate TS AST mirror types in `packages/story-dsl` so the desktop UI and any future Phase 5 CLI share one source of truth.

Purpose: Every subsequent phase depends on a stable, testable DSL parser. DSL-07 locks this crate as pure (no Tauri) so it can be consumed by a future headless CLI (Phase 5). DSL-05 and DSL-06 are the UX foundations that prevent the "cryptic pest error" failure mode documented in PITFALLS.md §9.

```rust
pub fn parse(source: &str) -> ParseResult;
pub fn parse_file(path: &std::path::Path) -> std::io::Result<ParseResult>;

pub struct ParseResult {
    pub ast: Option<Story>,              // None only if totally unrecoverable
    pub diagnostics: Vec<Diagnostic>,    // errors + warnings; ast may be present with errors
}

pub struct Story { pub meta: Meta, pub scenes: Vec<Scene>, pub span: Span }
pub struct Meta { pub app: Option<String>, pub viewport: Option<Viewport>, pub theme: Option<Theme>, pub speed: Option<f32>, pub span: Span }
pub struct Scene { pub name: String, pub commands: Vec<Command>, pub span: Span }

pub enum Command {
    Navigate { url: String, span: Span },
    Click { target: SelectorOrText, span: Span },
    Type { target: SelectorOrText, text: String, span: Span },
    Scroll { direction: ScrollDir, amount: Option<f32>, span: Span },
    Hover { target: SelectorOrText, span: Span },
    Drag { from: SelectorOrText, to: SelectorOrText, span: Span },
    Select { target: SelectorOrText, value: String, span: Span },
    Upload { target: SelectorOrText, path: String, span: Span },
    Wait { duration_ms: u64, span: Span },
    WaitFor { target: WaitForTarget, timeout_ms: Option<u64>, span: Span },
    Assert { target: AssertTarget, span: Span },
    Screenshot { name: String, span: Span },
    Pause { span: Span },
}

pub struct Span { pub start: usize, pub end: usize, pub line: u32, pub col: u32 }
pub struct Diagnostic { pub severity: Severity, pub message: String, pub span: Span, pub suggestion: Option<String> }
pub enum Severity { Error, Warning, Info }
```

    ```pest
    WHITESPACE = _{ " " | "\t" }
    COMMENT = _{ "#" ~ (!NEWLINE ~ ANY)* | "/*" ~ (!"*/" ~ ANY)* ~ "*/" }
    NEWLINE = _{ "\r\n" | "\n" | "\r" }
    EOI_OR_NL = _{ NEWLINE+ | EOI }

    story = { SOI ~ NEWLINE* ~ meta_block? ~ NEWLINE* ~ scene_block+ ~ NEWLINE* ~ EOI }

    meta_block = { "meta" ~ "{" ~ NEWLINE+ ~ meta_entry* ~ "}" ~ NEWLINE* }
    meta_entry = { meta_key ~ ":" ~ meta_value ~ EOI_OR_NL }
    meta_key = @{ ASCII_ALPHA ~ (ASCII_ALPHANUMERIC | "_")* }
    meta_value = { string | number | ident | viewport_tuple }
    viewport_tuple = { "{" ~ "width" ~ ":" ~ number ~ "," ~ "height" ~ ":" ~ number ~ "}" }

    scene_block = { "scene" ~ string ~ "{" ~ NEWLINE+ ~ statement* ~ "}" ~ NEWLINE* }
    statement = { command ~ EOI_OR_NL | recovery_line }
    recovery_line = @{ (!NEWLINE ~ ANY)+ }  // lenient — matched by layer 2 as "unknown"

    command = {
        cmd_navigate | cmd_click | cmd_type | cmd_scroll | cmd_hover | cmd_drag
      | cmd_select | cmd_upload | cmd_wait_for | cmd_wait | cmd_assert
      | cmd_screenshot | cmd_pause

    cmd_type = { "type" ~ target ~ string }
    cmd_scroll = { "scroll" ~ direction ~ number? }
    cmd_hover = { "hover" ~ target }
    cmd_drag = { "drag" ~ target ~ "to" ~ target }
    cmd_select = { "select" ~ target ~ string }

    target_selector = { "selector" ~ string }
    target_testid = { "testid" ~ string }
    target_aria = { "aria" ~ string }

    direction = { "up" | "down" | "left" | "right" }
    duration = @{ number ~ ("ms" | "s" | "m") }
    string = @{ "\"" ~ (("\\" ~ ANY) | (!"\"" ~ ANY))* ~ "\"" }
    ```

    ```rust
    use strsim::levenshtein;
    pub const KNOWN_VERBS: &[&str] = &["navigate","click","type","scroll","hover","drag","select","upload","wait","wait-for","assert","screenshot","pause"];
    pub const KNOWN_META_KEYS: &[&str] = &["app","viewport","theme","speed"];
    pub fn did_you_mean(input: &str, candidates: &[&str]) -> Option<String> {
        candidates.iter()
            .map(|c| (c, levenshtein(input, c)))
            .filter(|(_, d)| *d <= 2)
            .min_by_key(|(_, d)| *d)
            .map(|(c, _)| c.to_string())
    }
    ```
