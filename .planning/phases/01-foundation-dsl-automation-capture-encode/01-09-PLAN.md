    import { invoke } from '@tauri-apps/api/core';
    import type { ParseResult, Diagnostic } from '@storycapture/shared-types';

    export const storyDiagnosticsLinter = linter(async (view) => {
      const source = view.state.doc.toString();
      const result = await invoke<ParseResult>('parse_story', { source });
      return result.diagnostics.map((d): CmDiagnostic => ({

        to: d.span.end,
        severity: d.severity === 'Error' ? 'error' : d.severity === 'Warning' ? 'warning' : 'info',
        message: d.suggestion ? `${d.message} (did you mean ${d.suggestion}?)` : d.message,
      }));
    }, { delay: 300 });
