       ```rust
       use crate::filters::QualityPreset;
       use crate::probe::HardwareEncoder;

       /// Compute pixel-based target bitrate in kbps, capped at 40_000.
       pub fn pixel_based_kbps(output_w: u32, output_h: u32) -> u32 {
           let raw = (output_w as u64).saturating_mul(output_h as u64)
               .saturating_mul(3) / 1000;
           raw.min(40_000) as u32
       }

       /// Resolve preset + encoder + output dims into FFmpeg argv flags for
       /// rate-control + speed-preset. Caller concatenates these with the
       /// -c:v / -pix_fmt / -vf / audio flags.
       pub fn resolve(
           preset: QualityPreset,
           encoder: HardwareEncoder,
           output_w: u32,
           output_h: u32,
       ) -> Vec<String>;
       ```
       Implementation: build `Vec<String>` with string literals + `format!("{}k", kbps)`. Use a helper `kbps_scaled(base: u32, numer: u32, denom: u32) -> u32` that computes `(base as u64 * numer as u64 / denom as u64).min(40_000) as u32` to avoid float arithmetic (reproducibility).
